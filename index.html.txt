<script>
const siteConfig = {
    siteName: "Seller Watermark Tool",
    copyright: "© Seller Tools",
    defaultWater: "Official Store",
    adminPassword: "admin123"
};
let imgList = [];
const zip = new JSZip();

document.getElementById("imgFile").addEventListener("change",e=>{
    imgList = Array.from(e.target.files);
    alert("Selected: " + imgList.length + " images");
});

document.getElementById("sizeTemplate").addEventListener("change",function(){
    const t = this.value;
    const ta = document.getElementById("sizeText");
    if(t==="clothes")ta.value="S: Bust 90cm\nM: Bust 96cm\nL: Bust 102cm\nXL: Bust 108cm";
    else if(t==="shoes")ta.value="EU36: 22.5cm\nEU37: 23.0cm\nEU38: 23.5cm";
    else if(t==="bag")ta.value="S:20*10*15cm\nM:25*12*18cm\nL:30*15*20cm";
    else ta.value="";
});

async function startBatchWater(){
    if(!imgList.length)return alert("Please upload images first");
    zip.root.children = {};
    document.getElementById("preview").style.display="none";
    for(let i=0;i<imgList.length;i++){
        await handleImg(imgList[i],i+1);
    }
    const blob = await zip.generateAsync({type:"blob"});
    saveAs(blob,"watermark_images.zip");
    alert("All done, ZIP downloaded");
}

function handleImg(file,idx){
    return new Promise(resolve=>{
        const r = new FileReader();
        r.onload = e=>draw(e.target.result,file.name,idx,resolve);
        r.readAsDataURL(file);
    });
}

function draw(src,name,idx,resolve){
    const c = document.createElement("canvas");
    const ctx = c.getContext("2d");
    const img = new Image();
    img.crossOrigin="anonymous";
    img.onload=()=>{
        c.width=img.width;c.height=img.height;
        ctx.drawImage(img,0,0);
        const text = document.getElementById("waterText").value||siteConfig.defaultWater;
        const color = document.getElementById("waterColor").value;
        const size = document.getElementById("waterSize").value;
        const pos = document.getElementById("waterPos").value;
        const opa = document.getElementById("waterOpacity").value;
        const quality = +document.getElementById("compressQuality").value;
        const fmt = document.getElementById("imgFormat").value;

        ctx.globalAlpha=opa;
        ctx.fillStyle=color;
        ctx.font=`${size}px sans-serif`;

        if(pos==="repeat"){
            for(let x=0;x<c.width;x+=120){
                for(let y=0;y<c.height;y+=60){
                    ctx.save();ctx.translate(x,y);ctx.rotate(-20*Math.PI/180);
                    ctx.fillText(text,0,0);ctx.restore();
                }
            }
        }else{
            let x=20,y=30;
            if(pos==="bottom-right"){x=c.width-120;y=c.height-20;}
            if(pos==="bottom-left"){x=20;y=c.height-20;}
            if(pos==="top-right"){x=c.width-120;y=40;}
            if(pos==="center"){x=c.width/2-40;y=c.height/2;}
            ctx.fillText(text,x,y);
        }
        const url = c.toDataURL(`image/${fmt}`,quality);
        if(idx===1){document.getElementById("preview").src=url;document.getElementById("preview").style.display="block";}
        zip.file(`img_${idx}.${fmt}`,url.split(",")[1],{base64:true});
        resolve();
    };
    img.src=src;
}

function createSizeChart(){
    const text = document.getElementById("sizeText").value;
    if(!text)return alert("Please fill size content");
    const c = document.createElement("canvas");
    const ctx = c.getContext("2d");
    c.width=500;c.height=350;
    ctx.fillStyle="#fff";ctx.fillRect(0,0,c.width,c.height);
    ctx.fillStyle="#333";ctx.font="18px sans-serif";
    text.split("\n").forEach((line,i)=>ctx.fillText(line,30,40+i*30));
    const fmt = document.getElementById("imgFormat").value;
    const url = c.toDataURL(`image/${fmt}`,0.8);
    document.getElementById("preview").src=url;
    document.getElementById("preview").style.display="block";
    const a=document.createElement("a");
    a.href=url;a.download=`size_chart.${fmt}`;a.click();
}

function showAdminLogin(){document.getElementById("adminMask").style.display="flex";}
function closeAdmin(){document.getElementById("adminMask").style.display="none";}
function closeAdminPanel(){document.getElementById("adminPanelMask").style.display="none";}
function checkAdminPwd(){
    const p = document.getElementById("adminPwd").value;
    if(p===siteConfig.adminPassword){
        closeAdmin();
        document.getElementById("siteNameInput").value=siteConfig.siteName;
        document.getElementById("defaultWaterInput").value=siteConfig.defaultWater;
        document.getElementById("copyrightInput").value=siteConfig.copyright;
        document.getElementById("adminPanelMask").style.display="flex";
    }else alert("Wrong password");
}
function saveAdminConfig(){
    siteConfig.siteName=document.getElementById("siteNameInput").value;
    siteConfig.defaultWater=document.getElementById("defaultWaterInput").value;
    siteConfig.copyright=document.getElementById("copyrightInput").value;
    alert("Saved");closeAdminPanel();
}
